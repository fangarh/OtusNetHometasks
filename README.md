```
    {

        public int SomeField1;

        public string SomeText { get; set; }
        public double SomeDouble { get; set; }

        public ClassForTests()
        {
            SomeField1 = 1;
            _someField2 = 3;
            SomeText = "hello";
            SomeDouble = 2.222;
        }

        public int GetSomeField2() => _someField2;
    }
```
